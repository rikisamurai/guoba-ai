## Object Utilities

### objectKeys

Type-safe `Object.keys`.

```ts
function objectKeys<T extends object>(obj: T): (keyof T)[]
```

```ts
objectKeys({ a: 1, b: 2 }) // ['a', 'b'] typed as ('a' | 'b')[]
```

### objectEntries

Type-safe `Object.entries`.

```ts
function objectEntries<T extends object>(obj: T): [keyof T, T[keyof T]][]
```

```ts
objectEntries({ a: 1, b: 2 }) // [['a', 1], ['b', 2]]
```

### deepClone

Deep clone a value using `structuredClone`.

```ts
function deepClone<T>(obj: T): T
```

```ts
const original = { a: { b: 1 } }
const copy = deepClone(original)
copy.a.b = 2
original.a.b // still 1
```

### omit

Create a new object with specified keys omitted.

```ts
function omit<T extends object, K extends keyof T>(obj: T, keys: K[]): Omit<T, K>
```

```ts
omit({ a: 1, b: 2, c: 3 }, ['b', 'c']) // { a: 1 }
```

### pick

Create a new object with only the specified keys.

```ts
function pick<T extends object, K extends keyof T>(obj: T, keys: K[]): Pick<T, K>
```

```ts
pick({ a: 1, b: 2, c: 3 }, ['a', 'c']) // { a: 1, c: 3 }
```

### deepMerge

Deep merge objects. Arrays are replaced, not merged. Does not mutate the target.

```ts
function deepMerge<T extends object>(target: T, ...sources: DeepPartial<T>[]): T
```

```ts
deepMerge({ a: 1, b: { c: 2 } }, { b: { d: 3 } })
// { a: 1, b: { c: 2, d: 3 } }
```
