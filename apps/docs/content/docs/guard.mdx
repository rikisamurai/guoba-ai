## Type Guards

### isString

Check if a value is a string.

```ts
function isString(val: unknown): val is string
```

```ts
isString('hello') // true
isString(123) // false
```

### isNumber

Check if a value is a number.

```ts
function isNumber(val: unknown): val is number
```

```ts
isNumber(42) // true
isNumber('42') // false
```

### isBoolean

Check if a value is a boolean.

```ts
function isBoolean(val: unknown): val is boolean
```

```ts
isBoolean(true) // true
isBoolean(0) // false
```

### isObject

Check if a value is a plain object (not an array or null).

```ts
function isObject(val: unknown): val is Record<string, any>
```

```ts
isObject({ a: 1 }) // true
isObject([1, 2]) // false
isObject(null) // false
```

### isFunction

Check if a value is a function.

```ts
function isFunction(val: unknown): val is Function
```

```ts
isFunction(() => {}) // true
isFunction({}) // false
```

### isDef

Check if a value is defined (not undefined).

```ts
function isDef<T>(val: T | undefined): val is T
```

```ts
isDef(0) // true
isDef(null) // true
isDef(undefined) // false
```

### isUndef

Check if a value is undefined.

```ts
function isUndef(val: unknown): val is undefined
```

### isNull

Check if a value is null.

```ts
function isNull(val: unknown): val is null
```

### isNullOrUndef

Check if a value is null or undefined.

```ts
function isNullOrUndef(val: unknown): val is null | undefined
```

### isEmpty

Check if a value is empty. Returns `true` for `null`, `undefined`, empty strings, empty arrays, and empty objects.

```ts
function isEmpty(val: unknown): boolean
```

```ts
isEmpty([]) // true
isEmpty({}) // true
isEmpty('') // true
isEmpty(null) // true
isEmpty([1]) // false
isEmpty(0) // false
```

### notNullish

Check if a value is not null or undefined. Useful as a filter predicate for type narrowing.

```ts
function notNullish<T>(val: T | null | undefined): val is NonNullable<T>
```

```ts
const values = [1, null, 2, undefined, 3]
const filtered = values.filter(notNullish) // [1, 2, 3] typed as number[]
```
